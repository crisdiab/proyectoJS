MODULES EXPORT AND IMPORT

A esto lo llamo un browserify nativo. Ahora JavaScript se empieza a parecer a lenguajes como Python o Ruby. Llamamos a las funciones desde los propios Scripts, sin tener que importarlos en el HTML, si usamos JavaScript en el navegador. 


//File: lib/person.js
module "person" {  
    export function hello(nombre) {
        return nombre;
    }
}
También se puede exportar así, export function hello(nombre) {...}; o también, si solo es una función la que tiene el módulo: export default function(nombre) {...}; 

//File: app.js
import { hello } from "person";  
var app = {  
    foo: function() {
        hello("Carlos");
    }
}
export app;

IMPORT EXPORT EN ES6 

//  lib/math.js
export function sum (x, y) { return x + y }
export var pi = 3.141593

//  someApp.js
import * as math from "lib/math"
console.log("2p = " + math.sum(math.pi, math.pi))

//  otherApp.js
import { sum, pi } from "lib/math"
console.log("2p = " + sum(pi, pi))
en ES5 era 

//  lib/math.js
LibMath = {};
LibMath.sum = function (x, y) { return x + y };
LibMath.pi = 3.141593;

//  someApp.js
var math = LibMath;
console.log("2p = " + math.sum(math.pi, math.pi));

//  otherApp.js
var sum = LibMath.sum, pi = LibMath.pi;
console.log("2p = " + sum(pi, pi));
