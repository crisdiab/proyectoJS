ES6 a empezado a dar soporte a las funciones GETTER y SETTER como lo podemos ver en el siguiente ejemplo


          class Empleado {

                constructor(nombre) {
                    this._nombre = nombre;
                }

                get nombre() {
                  if(this._nombre) {
                    return 'SR. ' + this._nombre.toUpperCase();  
                  } else {
                    return undefined;
                  }  
                }

                set name(nuevoNombre) {
                  if (nuevoNombre == this._nombre) {
                    console.log('Actualmente tiene ese nombre.');
                  } else if (nuevoNombre) {
                    this._nombre = nuevoNombre;
                  } else {
                    return false;
                  }
                }
            }
